class Solution {
    public int firstUniqueEven(int[] nums) {
        for(int i=0;i<nums.length;i++){
            if(nums[i]%2==0){
                if(check(nums,i)){
                    return nums[i];
                }
            }
        }
        return -1;
        
    }

    boolean check(int[] arr,int ind){
        for(int i=0;i<arr.length;i++){
            if(ind!=i && arr[i]==arr[ind]){
                return false;
            }
        }
        return true;
    }
}
