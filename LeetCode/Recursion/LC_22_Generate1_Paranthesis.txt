class Solution {
    public List<String> generateParenthesis(int n) {
        List<String> res=new ArrayList<>();
        generateValidParanthesis("",0,0,res,n);
        return res;
    }

    void generateValidParanthesis(String curr,int open,int close,List<String> res,int n){
        if(curr.length()==2*n){
            res.add(curr);
            return;
        }

        if(open<n){
            generateValidParanthesis(curr+"(",open+1,close,res,n);
        }
        if(close<open){
             generateValidParanthesis(curr+")",open,close+1,res,n);
        }
    }
}
