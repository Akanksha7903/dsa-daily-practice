/*
// Definition for a Node.
class Node {
    int val;
    Node next;
    Node random;

    public Node(int val) {
        this.val = val;
        this.next = null;
        this.random = null;
    }
}
*/

class Solution {
    public Node copyRandomList(Node head) {
        //Create the copy of each node and connect them in between
        Node temp=head;
        while(temp!=null){
            Node copyNode=new Node(temp.val);
            copyNode.next=temp.next;
            temp.next=copyNode;
            temp=temp.next.next;
        }
        
        //Connect the copy node random pointers
        temp=head;
        while(temp!=null){
            Node res=temp.next;
            if(temp.random!=null){
                res.random=temp.random.next;
            }else{
                res.random=null;
            }
            temp=temp.next.next;
        }
        

        //extract the copy node from the original node and return
        Node dummy=new Node(-1);
        temp=head;
        Node copy=dummy;

        while(temp!=null){
            Node res=temp.next;
            copy.next=res;
            temp.next=res.next;
            temp=temp.next;
            copy=copy.next;
        }
        return dummy.next;
        
    }
}
